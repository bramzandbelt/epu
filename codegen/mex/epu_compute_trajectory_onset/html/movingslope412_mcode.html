1function Dvec = movingslope(vec,supportlength,modelorder,dt)
  2% movingslope: estimate local slope for a sequence of points, using a sliding window
  3% usage: Dvec = movingslope(vec)
  4% usage: Dvec = movingslope(vec,supportlength)
  5% usage: Dvec = movingslope(vec,supportlength,modelorder)
  6% usage: Dvec = movingslope(vec,supportlength,modelorder,dt)
  7%
  8%
  9% movingslope uses filter to determine the slope of a curve stored
 10% as an equally (unit) spaced sequence of points. A patch is applied
 11% at each end where filter will have problems. A non-unit spacing
 12% can be supplied.
 13%
 14% Note that with a 3 point window and equally spaced data sequence,
 15% this code should be similar to gradient. However, with wider
 16% windows this tool will be more robust to noisy data sequences.
 17%
 18%
 19% arguments: (input)
 20%  vec - row of column vector, to be differentiated. vec must be of
 21%        length at least 2.
 22%
 23%  supportlength - (OPTIONAL) scalar integer - defines the number of
 24%        points used for the moving window. supportlength may be no
 25%        more than the length of vec.
 26%
 27%        supportlength must be at least 2, but no more than length(vec)
 28%
 29%        If supportlength is an odd number, then the sliding window
 30%        will be central. If it is an even number, then the window
 31%        will be slid backwards by one element. Thus a 2 point window
 32%        will result in a backwards differences used, except at the
 33%        very first point, where a forward difference will be used.
 34%
 35%        DEFAULT: supportlength = 3
 36%
 37%  modelorder - (OPTIONAL) - scalar - Defines the order of the windowed
 38%        model used to estimate the slope. When model order is 1, the
 39%        model is a linear one. If modelorder is less than supportlength-1.
 40%        then the sliding window will be a regression one. If modelorder
 41%        is equal to supportlength-1, then the window will result in a
 42%        sliding Lagrange interpolant.
 43%
 44%        modelorder must be at least 1, but not exceeding
 45%        min(10,supportlength-1)
 46%
 47%        DEFAULT: modelorder = 1
 48%
 49%  dt - (OPTIONAL) - scalar - spacing for sequences which do not have
 50%        a unit spacing.
 51%
 52%        DEFAULT: dt = 1
 53%
 54% arguments: (output)
 55%  Dvec = vector of derivative estimates, Dvec will be of the same size
 56%        and shape as is vec.
 57% 
 58%
 59% Example:
 60%  Estimate the first derivative using a 7 point window with first through
 61%  fourth order models in the sliding window. Note that the higher order
 62%  approximations provide better accuracy on this curve with no noise.
 63%  
 64%  t = 0:.1:1;
 65%  vec = exp(t);
 66%
 67%  Dvec = movingslope(vec,7,1,.1)
 68%  Dvec =
 69%  Columns 1 through 7
 70%    1.3657  1.3657  1.3657  1.3657  1.5093  1.668  1.8435
 71%  Columns 8 through 11
 72%    2.0373  2.0373  2.0373  2.0373
 73%
 74%  Dvec = movingslope(vec,7,2,.1)
 75%  Dvec =
 76%  Columns 1 through 7
 77%    0.95747 1.0935  1.2296  1.3657  1.5093  1.668  1.8435
 78%  Columns 8 through 11
 79%    2.0373  2.2403  2.4433  2.6463
 80%
 81%  Dvec = movingslope(vec,7,3,.1)
 82%  Dvec =
 83%  Columns 1 through 7
 84%    1.0027  1.1049  1.2206  1.3498  1.4918  1.6487  1.8221
 85%  Columns 8 through 11
 86%    2.0137  2.2268  2.4602  2.7138
 87%
 88%  Dvec = movingslope(vec,7,4,.1)
 89%  Dvec =
 90%    Columns 1 through 7
 91%    0.99988 1.1052  1.2214  1.3498  1.4918  1.6487  1.8221
 92%  Columns 8 through 11
 93%    2.0137  2.2255  2.4597  2.7181
 94%
 95%
 96% Example:
 97%  Estimate the slope of a noisy curve, using a locally quadratic
 98%  approximation. In this case, use a straight line so that we know
 99%  the true slope should be 1. Use a wide window, since we have
100%  noisy data.
101%  
102%  t = 0:100;
103%  vec = t + randn(size(t));
104%  Dvec = movingslope(vec,10,2,1)
105%  mean(Dvec)
106%  ans = 
107%     1.0013
108%  std(Dvec)
109%  ans =
110%     0.10598
111%
112%  By way of comparison, gradient gives a much noisier estimate
113%  of the slope of this curve.
114%
115%  std(gradient(vec))
116%  ans =
117%     0.69847
118%
119%
120% Example:
121%  As a time test, generate random data vector of length 500000.
122%  Compute the slopes using a window of width 10.
123%
124%  vec = rand(1,500000);
125%  tic
126%  Dvec = movingslope(vec,10,2);
127%  toc
128%
129%  Elapsed time is 0.626021 seconds.
130%
131%
132% See also: gradient
133%
134%
135% Author: John D'Errico
136% e-mail: woodchips@rochester.rr.com
137% Release: 1.0
138% Release date: 10/19/07
139
140% how long is vec? is it a vector?
141if (nargin==0)
142  help movingslope
143  return
144end
145if ~isvector(vec)
146  error('vec must be a row or column vector')
147end
148n = length(vec);
149
150% supply defaults
151if (nargin<4) || isempty(dt)
152  dt = 1;
153end
154if (nargin<3) || isempty(modelorder)
155  modelorder = 1;
156end
157if (nargin<2) || isempty(supportlength)
158  supportlength = 3;
159end
160
161% check the parameters for problems
162if (length(supportlength)~=1) || (supportlength<=1) || (supportlength>n) || (supportlength~=floor(supportlength))
163  error('supportlength must be a scalar integer, >= 2, and no more than length(vec)')
164end
165if (length(modelorder)~=1) || (modelorder<1) || (modelorder>min(10,supportlength-1)) || (modelorder~=floor(modelorder))
166  error('modelorder must be a scalar integer, >= 1, and no more than min(10,supportlength-1)')
167end
168if (length(dt)~=1) || (dt<0)
169  error('dt must be a positive scalar numeric variable')
170end
171
172% now build the filter coefficients to estimate the slope
173if mod(supportlength,2) == 1
174  parity = 1; % odd parity
175else
176  parity = 0;
177end
178s = (supportlength-parity)/2;
179t = ((-s+1-parity):s)';
180coef = getcoef(t,supportlength,modelorder);
181
182% Apply the filter to the entire vector
183f = filter(-coef,1,vec);
184Dvec = zeros(size(vec));
185Dvec(s+(1:(n-supportlength+1))) = f(supportlength:end);
186
187% patch each end
188vec = vec(:);
189for i = 1:s
190  % patch the first few points
191  t = (1:supportlength)' - i;
192  coef = getcoef(t,supportlength,modelorder);
193  
194  Dvec(i) = coef*vec(1:supportlength);
195  
196  % patch the end points
197  if i<(s + parity)
198    t = (1:supportlength)' - supportlength + i - 1;
199    coef = getcoef(t,supportlength,modelorder);
200    Dvec(n - i + 1) = coef*vec(n + (0:(supportlength-1)) + 1 - supportlength);
201  end
202end
203
204% scale by the supplied spacing
205Dvec = Dvec/dt;
206% all done
207
208end % mainline end
209

211% subfunction, used to compute the filter coefficients
